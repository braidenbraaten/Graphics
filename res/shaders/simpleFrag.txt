#version 150

//This is GLSL!

in vec4 vPosition;
in vec4 vColor;
out vec4 outColor;

uniform float time = 0.0;

void main() 
{

	outColor = vColor;
	outColor.r = (cos(time) + 1.0)/2.0;
	outColor.b = (sin(time) + 1.0)/2.0;
	outColor.a = (sin(time) + 1.0)/2.0;
	
}
